## 基础 AudioManager 框架

```typescript
// client/src/audio/AudioManager.ts
import { Howl } from 'howler';

export class AudioManager {
  private sounds: Map<string, Howl> = new Map();
  private enabled = true;
  private volume = 0.7;

  async init() {
    this.preloadSound('battle:card-trigger', '/assets/sounds/battle/card-trigger.mp3');
    this.preloadSound('battle:damage', '/assets/sounds/battle/damage.mp3');
    // ... 更多音效
  }

  private preloadSound(key: string, src: string) {
    const howl = new Howl({ src });
    this.sounds.set(key, howl);
  }

  playEffect(key: string) {
    if (!this.enabled) return;
    const sound = this.sounds.get(key);
    if (sound) {
      sound.volume(this.volume);
      sound.play();
    }
  }

  setVolume(vol: number) {
    this.volume = Math.clamp(vol, 0, 1);
  }

  toggle() {
    this.enabled = !this.enabled;
  }
}
```
