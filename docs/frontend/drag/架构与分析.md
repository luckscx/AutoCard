# AutoCard 拖拽系统 - 完整架构参考

## Executive Summary

AutoCard 拖拽系统实现了**协作式跨容器物品移动**，结合 Pixi.js 实现实时视觉反馈。三个组件协同工作：

1. **BoardRow.ts** - 底层拖拽机制（虚影、高亮、合成检测）
2. **Scene 类** - 高层逻辑（API 调用、跨容器协调）
3. **布局系统** - 网格数学（槽位映射、坐标转换）

```typescript
// 1. 启动（BoardRow.setupDrag - 第 107-133 行）
card.on('pointerdown', (e: FederatedPointerEvent) => {
  this.dragItem = item;
  this.dragging = card;
  const local = this.toLocal(e.global);
  this.dragOffset.x = local.x - card.x;  // ← 关键：计算偏移
  this.dragOffset.y = local.y - card.y;
  
  this.dragGhost = new CardView(item);   // ← 创建虚影
  this.dragGhost.alpha = 0.7;
  this.dragGhost.x = card.x;
  this.dragGhost.y = card.y;
  this.cardsLayer.addChild(this.dragGhost);  // ← 加入层级（渲染在最上）
  
  card.alpha = 0.3;
  card.cursor = 'grabbing';
  
  stage.on('pointermove', this.handleDragMove);  // ← 绑定 stage 处理器
  stage.on('pointerup', this.handleDragEnd);
  stage.on('pointerupoutside', this.handleDragEnd);
});

// 2. 追踪（BoardRow.handleDragMove - 第 135-149 行）
private handleDragMove = (e: FederatedPointerEvent) => {
  if (!this.dragGhost || !this.dragItem) return;
  
  const local = this.toLocal(e.global);             // 全局 → 局部
  this.dragGhost.x = local.x - this.dragOffset.x;   // ← 关键：虚影位置公式
  this.dragGhost.y = local.y - this.dragOffset.y;
  
  if (this.isWithinOwnRow(e.global.y)) {
    const slotIdx = this.getSlotAtLocal(local.x);
    this.showHighlight(slotIdx, this.dragItem.size);
  } else {
    this.clearHighlight();
  }
  
  this.onDragging?.(this.dragItem, e.global.x, e.global.y);  // ← 场景回调
};

// 3. 视觉反馈（ShopScene - 第 196-211 行）
this.boardRow.onDragging = (_item, gx, gy) => {
  if (this.stashOpen) {
    if (gy < Z2_Y) {
      this.sellHint.visible = true;
      this.stashRow.clearExternalHighlight();
    } else if (gy >= Z2_Y && gy < Z2_Y + Z2_H) {
      this.sellHint.visible = false;
      this.stashRow.showExternalHighlight(gx, _item.size);  // ← 跨容器反馈
    }
  } else {
    this.sellHint.visible = gy < Z3_Y;
  }
};

// 4. 释放（BoardRow.handleDragEnd - 第 151-196 行）
private handleDragEnd = (e: FederatedPointerEvent) => {
  stage.off('pointermove', this.handleDragMove);
  stage.off('pointerup', this.handleDragEnd);
  
  if (this.dragging) {
    this.dragging.alpha = 1;
    this.dragging.cursor = 'grab';
  }
  
  if (this.dragGhost) {
    this.cardsLayer.removeChild(this.dragGhost);
    this.dragGhost.destroy();
    this.dragGhost = null;
  }
  
  const item = this.dragItem;
  this.dragItem = null;
  
  if (item) {
    if (this.isWithinOwnRow(e.global.y)) {
      const targetSlot = this.getSlotAtLocal(...);
      const targetItem = this.findItemCoveringSlot(targetSlot);
      
      if (targetItem && targetItem.itemId === item.itemId && targetItem.tier === item.tier) {
        this.onMerge?.(item, targetItem);
      } else if (targetSlot >= 0) {
        this.onSwap?.(item, targetSlot);
      }
    } else {
      this.onDragOut?.(item, e.global.x, e.global.y);
    }
  }
  
  this.onDragStop?.();
};

// 5. API 调用（ShopScene - 第 317-327 行）
private async handleCrossMoveToStash(item: SlotItem, toSlot: number) {
  try {
    const result = await api.placeItem(
      gameState.run!.id,
      'board', item.slotIndex,
      'stash', toSlot
    );
    gameState.setRun(result.run);
    this.boardRow.update(result.run.board);
    this.stashRow.update(result.run.stash);
    this.bottomBar.update(result.run);
  } catch (e: any) {
    console.error('Move to stash failed:', e.message);
  }
}

// 6. UI 更新（BoardRow.update - 第 61-74 行）
update(items: SlotItem[]) {
  this.items = [...items];
  this.cardsLayer.removeChildren();
  
  for (const item of items) {
    const card = new CardView(item);
    card.x = item.slotIndex * (CARD_UNIT + CARD_GAP);
    card.y = 0;
    card.eventMode = 'static';
    card.cursor = 'grab';
    this.setupDrag(card, item);
    this.cardsLayer.addChild(card);
  }
}
```

### 拖拽流程详解

#### 1️⃣ 鼠标按下 (pointerdown)

```typescript
// 第 108 行 - setupDrag 方法
private setupDrag(card: CardView, item: SlotItem) {
  card.on('pointerdown', (e: FederatedPointerEvent) => {
    this.clearSlotTargetGlow();
    this.dragItem = item;
    this.dragging = card;
    
    const local = this.toLocal(e.global);
    this.dragOffset.x = local.x - card.x;
    this.dragOffset.y = local.y - card.y;

    this.dragGhost.alpha = 0.7;
    this.dragGhost.x = card.x;
    this.dragGhost.y = card.y;
    this.cardsLayer.addChild(this.dragGhost);

    card.alpha = 0.3;
    card.cursor = 'grabbing';

    const stage = this.getStage();
    if (stage) {
      stage.eventMode = 'static';
      stage.on('pointermove', this.handleDragMove);
      stage.on('pointerup', this.handleDragEnd);
      stage.on('pointerupoutside', this.handleDragEnd);

}
```

```typescript
// 第 222-227 行
private getSlotAtLocal(localX: number): number {
```

## 8. Common Drag Errors & Solutions

| Error | Cause | Fix |
|-------|-------|-----|
| Ghost disappears mid-drag | `dragGhost = null` too early | Call `destroy()` first |
| Cursor stays "grabbing" | Original card not restored | Verify `alpha=1` and `cursor='grab'` in handleDragEnd |
| Highlight doesn't follow | `onDragging` not firing | Check callback binding |
| Cards freeze during drag | Stage events not unbound | Check `stage.off()` calls |
| Ghost offset incorrect | `dragOffset` wrong | Verify: `dragOffset = mouse_local - card_origin` |
| Cross-container fails | `toLocal()` wrong container | `toLocal()` is relative to THIS container |
| Merge doesn't trigger | `findItemCoveringSlot()` returns null | Check slot calculation |
